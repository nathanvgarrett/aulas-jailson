A <meta>tag define metadados sobre um documento HTML. Metadados são dados (informações) sobre dados.

<head>
  <meta charset="UTF-8"> <!-- Formatação do texto - padrão de acentuação --> 
  <!-- tag | atributo | valor -->
  <meta name="description" content="Free Web tutorials"> 
  <meta name="keywords" content="HTML, CSS, JavaScript"> <!-- palavras-chaves -->
  <meta name="author" content="John Doe"> <!-- Empresa | Nathanzin -->
  <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- responsibilidades -->
</head>
